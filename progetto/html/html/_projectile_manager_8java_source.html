122  public void newProjectile(Tower t, Enemy e) {
123  int type = getProjectileType(t);
124 
125  int distx = (int) (t.getX() - e.getX());
126  int disty = (int) (t.getY() - e.getY());
127  int totaldistance = Math.abs(distx) + Math.abs(disty);
128 
129  float xper = (float) Math.abs(distx) / totaldistance;
130  float yper = 1.0f - xper;
131  float speedx = xper * helpz.Constants.Projectiles.getSpeed(type);
132  float speedy = helpz.Constants.Projectiles.getSpeed(type) - speedx;
133 
134  if (t.getX() > e.getX()) {
135  speedx *= -1;
136  }
137  if (t.getY() > e.getY()) {
138  speedy *= -1;
139  }
140 
141  float rotate = 0;
142 
143  if (t.getTypetower() != SAKURA && t.getTypetower() != AKAINU && t.getTypetower() != HEAD) {
144  float arc = (float) Math.atan(disty / (float) distx);
145  rotate = (float) Math.toDegrees(arc);
146  if (distx < 0) {
147  rotate += 180;
148  }
149  }
150  synchronized (a) {
151  projectiles.add(new Projectile(t.getX() + Tile.spriteWidth / 2, t.getY() + Tile.spriteHeight / 2, speedx, speedy, t.getDamage(), rotate, projId++, type));
152  }
